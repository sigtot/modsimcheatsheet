%------------ Skew symmetric matrix ---------------
\begin{tikzpicture}
\node [mybox] (box){%
    \begin{minipage}{0.3\textwidth}
        \begin{align*}
            \mathbf{a} = \begin{bmatrix}
                a_1 \\
                a_2 \\
                a_3 
            \end{bmatrix}
            \iff
            \mathbf{a}^\times = \begin{bmatrix}
                0 & -a_3 & a_2 \\
                a_3 & 0 & -a_1 \\
                -a_2 & a_1 & 0
            \end{bmatrix}
        \end{align*}
        \begin{itemize}
            \item $u \times v$ corresponds to $\mathbf{u}^\times \mathbf{v}$
            \item $(\mathbf a^\times)^\top = -(\mathbf a^\times)$
        \end{itemize}
    \end{minipage}
};

%------------ Homogenous Transformation Matrix ---------------
\begin{tikzpicture}
\node [mybox] (box){%
    \begin{minipage}{0.3\textwidth}
        \begin{align*}
        \mathbf T ^a_b = \begin{bmatrix}
            \mathbf R ^ a _ b & \mathbf r ^ a _ {ab} \\
            \mathbf 0 ^ \top & 1
        \end{bmatrix}
        \in SE(3)
        \end{align*}
        \begin{align*}
        (\mathbf T ^a_b)^{-1} = \begin{bmatrix}
            (\mathbf R^a_b)^\top & -(\mathbf R ^a_b)^\top \mathbf r^a_{ab} \\
            \mathbf 0^\top & 1
        \end{bmatrix}
        = \begin{bmatrix}
            \mathbf R ^b_a & \mathbf r ^a_{ba} \\
            \mathbf 0 ^\top & 1
        \end{bmatrix}
        = \mathbf T^b_a
        \end{align*}
Note that the positon of the a-frame in the transformed frame (the b-frame) is
$\mathbf r^a_{ba} = -(\mathbf R^a_b)^\top \mathbf r^a_{ab}$.
    \end{minipage}
};
\node[fancytitle, right=10pt] at (box.north west) {Homogenous Transformation Matrix};
\end{tikzpicture}
